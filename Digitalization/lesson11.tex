\Title{Numbers Plural Case}

% Section 58
\newSection The nominative plural case ends in \m{-d}, which is added to the genitive singular.

	\threeColumnsTable
	\n{Nominative Singular} 	& \n{Genitive Singular} 	& \n{Nominative Plural} \\
	linn	`town'				& \m{linna}					& \m{linna/d}	`towns' \\
	lind	`bird'				& \m{linnu}					& \m{linnu/d}	`birds' \\
	riik	`state'				& \m{riigi}					& \m{riigi/d}	`states' \\
	lipp	`flag'				& \m{lipu}					& \m{lipu/d}	`flags' \\
	laps	`child'				& \m{lapse}					& \m{lapse/d}	`children' \\
	vend	`brother'			& \m{venna}					& \m{venna/d}	`brothers' \\
	õde		`sister'			& \m{õe}					& \m{õe/d}		`sisters' \\
	vana	`old'				& \m{vana}					& \m{vana/d}	`old (ones)' \\
	noor	`young'				& \m{noore}					& \m{noore/d}	`young (ones)' \\
	ilus	`pretty'			& \m{ilusa}					& \m{ilusa/d}	`pretty (ones)' \\
	armas	`dear'				& \m{armsa}					& \m{armsa/d}	`dear (ones)'
	\tableEnd

\m{Linnud} laulavad `Birds sing'. \m{Lapsed} mängivad `Children play'.

% Section 59
\newSection If a noun in the nominative plural case is modified by an adjective, the latter is also in the nominative plural:

	\twoColumnsTable
	\m{suured} linnad 	& `big towns' \\
	\m{vanad} inimesed 	& `old people' \\
	\m{armsad} õed 		& `dear sisters'
	\tableEnd

% Section 60
\newSection An adjective which is the predicate complement after the verb \m{on} `are' and a plural subject takes the nominative plural case.

	\twoColumnsTable
	Ajad on \m{rasked}.		&	`Times are hard.' \\
	Raamatud on \m{kallid}.	&	`Books are expensive.' \\
	Meie oleme \m{noored}.	&	`We are young.'
	\tableEnd
