% Fancy sections' symbol
\newcounter{fancysection}
\newcommand{\newSection}
{
	\stepcounter{fancysection} 
	\vspace*{1.5em}\textbf{\textsection\hspace*{0.3em}\arabic{fancysection}.}\hspace*{0.5em}
}

% Fancy lessons' title
\newcounter{fancylesson}
\newcommand{\newLesson}
{
	\stepcounter{fancylesson} 
	\newpage
	\begin{center}
		\textbf{{\huge Lesson\hspace*{0.5em}\arabic{fancylesson}}}
	\end{center}
}
